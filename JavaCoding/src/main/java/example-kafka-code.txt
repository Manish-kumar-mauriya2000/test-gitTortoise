@Component
public class PollingService {

    private int emptyPolls;

    public boolean shouldShutdown(
            boolean recordsFound) {

        if(recordsFound) {
            emptyPolls = 0;
            return false;
        }

        emptyPolls++;

        return emptyPolls >= 10;
    }
}


import java.util.List;

@Service
public class OffsetSeekService {

    public void seek(
            KafkaConsumer<String,String> consumer,
            String topic,
            int partition,
            long offset) {

        TopicPartition tp =
                new TopicPartition(
                        topic,
                        partition);

        consumer.assign(List.of(tp));

        consumer.seek(tp, offset);
    }
}

@Service
public class TimestampSeekService {

    public void seekToTimestamp(
            KafkaConsumer<String,String> consumer,
            String topic,
            int partition,
            long timestamp) {

        TopicPartition tp =
                new TopicPartition(topic,
                        partition);

        consumer.assign(List.of(tp));

        Map<TopicPartition, Long> request =
                Map.of(tp, timestamp);

        Map<TopicPartition,
                OffsetAndTimestamp>
                result =
                consumer.offsetsForTimes(
                        request);

        OffsetAndTimestamp oat =
                result.get(tp);

        if(oat != null) {

            consumer.seek(tp,
                    oat.offset());
        }
    }
}
